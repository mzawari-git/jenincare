🧩 1. Create the Job: ProcessSkinScan

📁 app/Jobs/ProcessSkinScan.php

<?php

namespace App\Jobs;

use App\Models\SkinScan;
use App\Services\AI\SkinAnalysisPipeline;
use Illuminate\Bus\Queueable;
use Illuminate\Contracts\Queue\ShouldQueue;
use Illuminate\Foundation\Bus\Dispatchable;

class ProcessSkinScan implements ShouldQueue
{
    use Dispatchable, Queueable;

    public $scanId;

    public function __construct($scanId)
    {
        $this->scanId = $scanId;
    }

    public function handle(SkinAnalysisPipeline $pipeline)
    {
        $scan = SkinScan::find($this->scanId);

        if (!$scan) return;

        // Update status
        $scan->update(['status' => 'processing']);

        try {
            $result = $pipeline->analyze($scan);

            $scan->update([
                'status' => 'completed',
                'analysis_result' => json_encode($result)
            ]);

        } catch (\Exception $e) {
            $scan->update([
                'status' => 'failed',
                'error' => $e->getMessage()
            ]);
        }
    }
}
🧠 2. Create the Pipeline Engine

📁 app/Services/AI/SkinAnalysisPipeline.php

<?php

namespace App\Services\AI;

use App\Services\AI\Providers\AIProviderFactory;

class SkinAnalysisPipeline
{
    public function analyze($scan)
    {
        // 1. Load image
        $imagePath = storage_path('app/' . $scan->image_path);

        // 2. Extract features (basic for now)
        $features = $this->extractFeatures($imagePath);

        // 3. Detect defects
        $defects = app(SkinDefectLibrary::class)->detect($features);

        // 4. Send to AI provider
        $provider = AIProviderFactory::make();

        $aiResponse = $provider->analyze([
            'features' => $features,
            'defects' => $defects
        ]);

        // 5. Return unified structure
        return [
            'features' => $features,
            'defects' => $defects,
            'ai' => $aiResponse,
            'score' => $this->calculateScore($defects),
        ];
    }

    private function extractFeatures($imagePath)
    {
        // Temporary simple logic (upgrade later)
        return [
            'brightness' => rand(40, 90),
            'redness' => rand(0, 100),
            'texture' => rand(0, 100),
            'oiliness' => rand(0, 100),
        ];
    }

    private function calculateScore($defects)
    {
        return max(0, 100 - count($defects) * 5);
    }
}
📚 3. Skin Defect Library (Core Intelligence)

📁 app/Services/AI/SkinDefectLibrary.php

<?php

namespace App\Services\AI;

class SkinDefectLibrary
{
    public function detect($features)
    {
        $defects = [];

        if ($features['redness'] > 70) {
            $defects[] = [
                'type' => 'erythema',
                'severity' => 8
            ];
        }

        if ($features['oiliness'] > 75) {
            $defects[] = [
                'type' => 'seborrhea',
                'severity' => 7
            ];
        }

        if ($features['texture'] > 60) {
            $defects[] = [
                'type' => 'rough_texture',
                'severity' => 6
            ];
        }

        return $defects;
    }
}
